Let \(\mathbf{u} =(u_1,\ldots,u_n)\text{,}\) \(\mathbf{v} =(v_1,\ldots,v_n)\text{,}\) \(\mathbf{w} =(w_1,\ldots,w_n)\) be vectors in \(\R^n\text{,}\) and let \(\alpha\in \R\text{.}\) For (2.9), we have

\begin{align*} \mathbf{u}\cdot (\mathbf{v}+\mathbf{w}) \amp = (u_1,u_2,\ldots,u_n) \cdot (v_1+w_1,v_2+w_2,\ldots,v_n+w_n)\\ \amp = (u_1(v_1+w_1),u_2(v_2+w_2),\ldots,u_n(v_n+w_n))\\ \amp = (u_1v_1+u_1w_1,u_2v_2+u_2w_2,\ldots,u_nv_n+u_nw_n)\\ \amp = (u_1v_1,u_2v_2,\ldots,u_nv_n)+(u_1w_1,u_2w_2,\ldots,u_nw_n)\\ \amp = \mathbf{u}\cdot\mathbf{v} + \mathbf{u}\cdot\mathbf{w}. \end{align*}

For the first of the two equalities in (2.10), we have

\begin{align*} \mathbf{u}\cdot (\alpha\mathbf{v}) \amp = (u_1,\ldots,u_n)\cdot (\alpha v_1,\ldots,\alpha v_n)\amp \text{(definition of scalar mult.)}\\ \amp = \sum_{i=1}^n u_i(\alpha v_i) \amp \text{(definition of dot prod.)}\\ \amp = \sum_{i=1}^n (u_i\alpha) v_i \amp \text{(mult. is associative)}\\ \amp = \sum_{i=1}^n (\alpha u_i) v_i \amp \text{(mult. is commutative)}\\ \amp = (\alpha u_1,\ldots,\alpha u_n)\cdot (v_1,\ldots,v_n)\amp \text{(definition of dot prod.)}\\ \amp = (\alpha \mathbf{u})\cdot \mathbf{v}\amp \text{(definition of scalar mult.)}. \end{align*}

For the second of the two equalities in (2.10), starting from the second to last expression in the above derivation, we have

\begin{equation*} \sum_{i=1}^n (\alpha u_i) v_i = \sum_{i=1}^n \alpha (u_i v_i) = \alpha \sum_{i=1}^n u_iv_i = \alpha (\mathbf{u}\cdot\mathbf{v}). \end{equation*}
