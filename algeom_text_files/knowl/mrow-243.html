\begin{align*} \text{Area } \amp = \int_0^{2\pi} \int_0^\alpha \frac{4\;dr\;d\theta}{(1+r^2)^2} \\ \amp = 2\int_0^{2\pi} \left[ \int_1^{1+\alpha^2} \frac{du}{u}\right] d\theta \;\;(\text{for }u=1+r^2)\\ \amp = 2\int_0^{2\pi}\frac{\alpha^2}{1+\alpha^2}d\theta\\ \amp = \frac{4\pi\alpha^2}{1+\alpha^2}. \end{align*}
