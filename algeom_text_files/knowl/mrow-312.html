\begin{align*} \cosh \left(\ln\left(\frac{1+u}{1-u}\right)\right) \amp = \frac{e^{\ln(\text{stuff})} + e^{-\ln(\text{stuff})}}{2} \;\; \left(\text{stuff }=\frac{1+u}{1-u}\right)\\ \amp = \frac{\frac{1+u}{1-u} + \frac{1-u}{1+u}}{2}\\ \amp = \frac{(1+u)^2 + (1-u)^2}{2(1-a^2)}\\ \amp = \frac{2(1+u^2)}{2(1-a^2)}\\ \amp = \frac{1+u^2}{1-u^2}. \end{align*}
