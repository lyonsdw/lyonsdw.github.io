\begin{align*} \cos(c) \amp= \frac{1-u^2}{1+u^2} \\ \amp = \frac{1-\left|\frac{B-A}{1+A^\ast B}\right|^2}{1+\left|\frac{B-A}{1+A^\ast B}\right|^2}\\ \amp = \frac{1-\frac{|A|^2+|B|^2}{1+|A|^2|B|^2}}{1+\frac{|A|^2+|B|^2}{1+|A|^2|B|^2}}\\ \amp = \frac{1-|A|^2 - |B|^2 + |A|^2|B|^2}{1+|A|^2 + |B|^2 + |A|^2|B|^2}\\ \amp = \frac{(1-|A|^2)(1 - |B|^2)}{(1+|A|^2)(1+ |B|^2)}\\ \amp = \cos(a)\cos(b). \end{align*}
